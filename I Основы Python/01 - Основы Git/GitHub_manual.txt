Remote repository - GitHub

#cоздаст локальную копию удалённого репозитория
git clone

#Большинство разработчиков именуют свои главные удалённые репозитории как origin
git checkout origin/master

#извлекать данные из удалённого репозитория
#Команда fetch скачивает все необходимые данные, потребуется вручную слить эти данные
git fetch

#чтобы отобразить все эти изменения
#pull = fetch + merg
#т.е. git fetch git merge o/master
git pull

#загрузка изменений в указанный удалённый репозиторий
#поведение команды git push без аргументов варьируется в зависимости от значения push.default
#зависит от версии git
#push.default=upstream
git push

#ЗАЛИТЬ ИЗМЕНЕНИЯ (ФИЧУ) В ИЗМЕНЕННЫЙ РЕПОЗИТОРИЙ
# 1. перебазирование (rebase)
#обновили наш локальный образ удалённого репозитория средствами git fetch. 
#перебазировали наши наработки, чтобы они отражали все изменения с удалённого репозитория
#опубликовали их с помощью git push
git fetch
git rebase o/master
git push
# 2 объединение (merge)
#обновили наше локальное представление удалённого репозитория с помощью git fetch
#объединили ваши новые наработки с нашими наработками (чтобы отразить изменения в удалённом репозитории) 
#опубликовали их с помощью git push.
git fetch
git merge o/master
git push
# или короче через pull
git pull --rebase
#или
git pull
git push
